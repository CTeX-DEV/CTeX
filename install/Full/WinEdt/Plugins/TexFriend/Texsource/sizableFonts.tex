\def\totxt#1{
\immediate\write\tf@txt{%
macro="\noexpand#1"}}
\def\ttotxt#1{
\immediate\write\tf@txt{%
macro="\string#1" \space package="stmaryrd"}}
\def\tttotxt#1{
\immediate\write\tf@txt{%
macro="\string#1"\space package="txfonts"}}
