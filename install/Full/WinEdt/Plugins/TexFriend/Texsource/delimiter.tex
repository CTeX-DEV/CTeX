\def\outputm{
 \begingroup
    \makeatletter
    \if@filesw
      \expandafter\newwrite\csname tf@txt\endcsname
      \immediate\openout \csname tf@txt\endcsname \jobname.txt\relax

  \endgroup
}
\def\totxt#1{
\immediate\write\tf@txt{%
macro="\noexpand#1"}}
\def\tttotxt#1{
\immediate\write\tf@txt{%
macro="\string#1" \space package="stmaryrd"}}
\def\ttotxt#1{
\immediate\write\tf@txt{%
macro="\string#1" \space package="amssymb"}}
 \makeatother
